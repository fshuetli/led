# Install WSGI
sudo apt-get install python-pip apache2 libapache2-mod-wsgi

# Activate WSGI
sudo a2enmod wsgi

# Add VirtualHost
sudo touch /etc/apache2/sites-available/LedApp.conf
########### add this content ##############
Listen 38090

<VirtualHost *:38090>
   WSGIDaemonProcess LedApp user=pi group=editorial threads=5 python-path=/home/pi/gits/led
   WSGIScriptAlias / /home/pi/gits/led/ledapp.wsgi
   <Directory /home/pi/gits/led>
      WSGIProcessGroup LedApp
      WSGIApplicationGroup %{GLOBAL}
      Order deny,allow
      Allow from all
   </Directory>
  <Files ledapp.wsgi>
    Require all granted
  </Files>
  ErrorLog ${APACHE_LOG_DIR}/error.log
  LogLevel warn
  CustomLog ${APACHE_LOG_DIR}/access.log combined
</VirtualHost>
###########

# Add App
sudo a2ensite LedApp
service apache2 reload

# Check Config
sudo apache2ctl configtest

# Start
sudo service apache2 restart

# Check for errors
tail -f /var/log/apache2/error.log
